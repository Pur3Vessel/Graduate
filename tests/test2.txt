
int main() {
    float a[3, 3] = {{1.1, 1.2, 2.3}, {5.5, 6.4, 3.3}, {1.7, 1.5, 5.9}};
    float b[3, 3] = {{1.1, 1.2, 2.3}, {5.5, 6.4, 3.3}, {1.7, 1.5, 5.9}};
    float c[3, 3];
    for (i = 0 to 3) {
        for (j = 0 to 3) {
            for (k = 0 to 3) {
                c[i, j] = c[i, j] + a[i, k] * b[k, j];
            }
        }
    }
    int d = 3;
    return c[d - 1, 2];
}
